Hello Git

sudo apt-get install git-core (Ubuntu/Debian)

Install Git on OS X:

curl -O http://kernel.org/pub/software/scm/git/git-1.7.2.2.tar.bz2
tar -jxvf git-1.7.2.2.tar.bz2
cd git-1.7.2.2
./configure --prefix=/usr/local
make
sudo make install
git --version (should return "git version 1.7.2.2")

Hello Ruby Version Manager (RVM)

nano ~/.bash_profile
[[ -s $HOME/.rvm/scripts/rvm ]] && source $HOME/.rvm/scripts/rvm
rvm install 1.9.2-preview3
rvm 1.9.2-preview3 --default
gem -v (should return 1.3.7)
ruby -v (should return ruby 1.9.2dev (2010-05-31 revision 28117) [x86_64-darwin10.4.0])
